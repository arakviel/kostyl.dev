public int Ackermann(int m, int n)
{
    if (m < 0 || n < 0)
        throw new ArgumentOutOfRangeException("m —ñ n –º–∞—é—Ç—å –±—É—Ç–∏ –Ω–µ–≤—ñ–¥'—î–º–Ω–∏–º–∏");

    int Compute(int mVal, int nVal)
    {
        if (mVal == 0)
            return nVal + 1;

        if (nVal == 0)
            return Compute(mVal - 1, 1);

        return Compute(mVal - 1, Compute(mVal, nVal - 1));
    }
}
