private static int Partition<T>(T[] array, int low, int high) where T : IComparable<T>
{
    T pivot = array[high];
    int i = low - 1;

    for (int j = low; j < high; j++)
    {
        if (T.Compare(array[j], pivot) < 0)
        {
            i++;
            (array[i], array[j]) = (array[j], array[i]);
        }
    }

    (array[i + 1], array[high]) = (array[high], array[i + 1]);
    return i + 1;
}
